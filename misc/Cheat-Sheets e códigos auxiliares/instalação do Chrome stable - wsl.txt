# Aparente nas distribuição do wsl apt install do google chrome roda um snap install por baixo dos panos 
# Como estamos num ambiente virtualizado, isso gera problemas e o webdriver não funciona corretamente.
# Os comando abaixo instalam o Chrome versão stable de forma compatível com o webdriver e Selenium



# Navega para pasta temporária
cd /tmp

# Baixa o pacote oficial do Google Chrome
wget https://dl.google.com/linux/direct/google-chrome-stable_current_amd64.deb

# Instala o Chrome
sudo dpkg -i google-chrome-stable_current_amd64.deb

# Resolve dependências faltantes (crucial!)
sudo apt install --fix-broken -y
